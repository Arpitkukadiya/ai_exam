SET-1 Questions & Answers 
% Square and Cube
square_cube(N, S, C) :- S is N*N, C is N*N*N.

% Max of 3 numbers
max3(A,B,C,Max) :- Max is max(A, max(B,C)).

% Sum 1 to N
sum_n(1,1).
sum_n(N,Sum) :- N > 1, N1 is N-1, sum_n(N1,Sum1), Sum is Sum1 + N.
SET-2 Questions & Answers 
% Circle area & perimeter
circle(R, Area, Perimeter) :- Area is 3.14*R*R, Perimeter is 2*3.14*R.

% Divisible by 4
divisible_by_4(N) :- 0 is N mod 4.

% Sum between N1 and N2
sum_range(N1,N1,N1).
sum_range(N1,N2,Sum) :- N1 < N2, N3 is N2-1, sum_range(N1,N3,Sum1), Sum is Sum1 + N2.
SET-3 Questions & Answers 
% Year to Minutes
year_to_minutes(Y, M) :- M is Y * 365 * 24 * 60.

% Odd or Even
is_even(N) :- 0 is N mod 2.
is_odd(N) :- 1 is N mod 2.

% Series from 1 to N
series(N) :- series_helper(1,N).
series_helper(I,N) :- I =< N, write(I), nl, I1 is I+1, series_helper(I1,N).
series_helper(I,N) :- I > N.

% Pass or Fail
result(Per,'Pass') :- Per >= 40.
result(Per,'Fail') :- Per < 40.

series(N) :- N > 0, write(N), nl, N1 is N-1, series(N1).
